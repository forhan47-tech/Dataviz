\section{Tables Examples}

% --- Basic Table ---
\subsection{Simple Table}
\begin{tabular}{|c|c|c|}
\hline
Column1 & Column2 & Column3 \\ \hline
Data1   & Data2   & Data3   \\ \hline
DataA   & DataB   & DataC   \\ \hline
\end{tabular}

% --- Alignment Example ---
\subsection{Alignment}
\begin{tabular}{lcr}
Left & Center & Right \\
A    & B      & C
\end{tabular}

% --- Table with Caption and Label ---
\subsection{Table with Caption}
\begin{table}[h]
\centering
\begin{tabular}{|c|c|}
\hline
Name & Age \\ \hline
Alice & 24 \\ \hline
Bob   & 30 \\ \hline
\end{tabular}
\caption{Sample Table with Names and Ages}
\label{tab:sample}
\end{table}
