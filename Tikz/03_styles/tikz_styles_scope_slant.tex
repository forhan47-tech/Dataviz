\documentclass{article}
\usepackage{tikz}
\begin{document}

\begin{tikzpicture}

  % --- Original square ---
  \draw (0,0) rectangle (1,1);

  % --- Slanted horizontally ---
  \begin{scope}[xslant=0.5]
    \draw[red] (2,0) rectangle (3,1);
  \end{scope}

  % --- Slanted vertically ---
  \begin{scope}[yslant=0.5]
    \draw[blue] (5,0) rectangle (6,1);
  \end{scope}

  % --- Combined slant (both directions) ---
  \begin{scope}[xslant=0.5, yslant=0.3]
    \draw[green] (8,0) rectangle (9,1);
  \end{scope}

\end{tikzpicture}

\end{document}
