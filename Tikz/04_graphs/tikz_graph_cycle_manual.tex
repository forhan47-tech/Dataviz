\documentclass{article}
\usepackage{tikz}
\usetikzlibrary{positioning} % enables relative placement

\begin{document}

\begin{tikzpicture}[nodes={draw,circle}]
  % --- Root node ---
  \node (A) {A};

  % --- Other nodes placed relative to A ---
  \node (B) [right=of A] {B};
  \node (C) [above=of B] {C};

  % --- Directed cycle edges ---
  \draw[->] (A) -- (B);
  \draw[->] (B) -- (C);
  \draw[->] (C) -- (A);
\end{tikzpicture}

\end{document}
