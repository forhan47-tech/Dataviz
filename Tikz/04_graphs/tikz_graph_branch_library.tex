\documentclass{article}
\usepackage{tikz}
\usetikzlibrary{graphs,positioning} % add positioning library

\begin{document}

\begin{tikzpicture}
  \graph[nodes={draw,circle}, edges={->}] {
    % --- Root node ---
    A;

    % --- Branch nodes placed relative to A ---
    B [below left=of A];
    C [below=of A];
    D [below right=of A];

    % --- Branching edges ---
    A -> {B, C, D};
  };
\end{tikzpicture}

\end{document}
