\documentclass[12pt]{article}

\begin{document}

\section{Lists Examples}

% --- Itemize (Bulleted List) ---
\subsection{Itemize}
\begin{itemize}
  \item First bullet
  \item Second bullet
  \item Third bullet
\end{itemize}

% --- Enumerate (Numbered List) ---
\subsection{Enumerate}
\begin{enumerate}
  \item Step one
  \item Step two
  \item Step three
\end{enumerate}

% --- Description (Definition List) ---
\subsection{Description}
\begin{description}
  \item[Term A:] Explanation of Term A
  \item[Term B:] Explanation of Term B
\end{description}

% --- Nested List Example ---
\subsection{Nested List}
\begin{enumerate}
  \item Step one
  \item Step two
    \begin{itemize}
      \item Sub-point A
      \item Sub-point B
    \end{itemize}
  \item Step three
\end{enumerate}
