\documentclass[12pt]{article}

\begin{document}

\section{Spacing Examples}

% --- Line Breaks ---
First line.\\
Second line.

% --- Paragraphs ---
This is the first paragraph.

This is the second paragraph. Notice the automatic indentation.

% --- Vertical Space ---
Text above.
\vspace{0.5cm} % adds half a cm vertical space
Text below.

% --- Horizontal Space ---
Word1\hspace{2cm}Word2\\
WordA\quad WordB\qquad WordC

% --- Indentation Control ---
\setlength{\parindent}{2em}
This paragraph starts with 2em indentation.
