\section{Cases and Piecewise Functions}

% --- Absolute Value Function ---

\[
|x| =
\begin{cases}
x  & \text{if } x \geq 0 \\
-x & \text{if } x < 0
\end{cases}
\]

% --- Custom Piecewise Function ---

\[
f(x) =
\begin{cases}
x^2 & \text{if } x \geq 0 \\
-x  & \text{if } x < 0
\end{cases}
\]
